# GitHub 仓库部署配置
# 请在下方填写您的信息

# 您的 GitHub 用户名
GITHUB_USERNAME="your-username"

# 仓库名称（推荐使用：your-username.github.io）
REPO_NAME="your-username.github.io"

# 完整的仓库 URL 将是：
# https://github.com/${GITHUB_USERNAME}/${REPO_NAME}.git

# 示例：
# GITHUB_USERNAME="zhangsan"
# REPO_NAME="zhangsan.github.io"
# 仓库地址：https://github.com/zhangsan/zhangsan.github.io.git
# 网站地址：https://zhangsan.github.io/

# 填写完成后保存此文件，然后运行：
# .\deploy.ps1
